pipeline {
    agent any

    stages {
        //stage 1: Collecting the code from github
        stage('git checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/spring-petclinic/spring-framework-petclinic.git'
            }
        }
        //stage 2: To run the tests
        stage('test') {
            steps {
                sh 'mvn clean test'
            }
        }
         //stage 3: running sonar analysis
        stage('code analysis') {
            steps {
               withSonarQubeEnv(installationName:'sonar',credentialsId: 'sonar') {
                sh 'mvn sonar:sonar'
                }
            }
        }
        //stage 4: To create the package
        stage('package') {
            steps {
                sh 'mvn package'
            }
        }
         //stage 5: To upload the artifact
        stage('artifact uploader') {
            steps {
                nexusArtifactUploader artifacts: [[artifactId: 'spring-framework-petclinic', classifier: '', file: 'target/petclinic.war', type: 'war']],
                credentialsId: 'nexus', groupId: 'org.springframework.samples', nexusUrl: '3.1.5.32:8081', nexusVersion: 'nexus3', protocol: 'http',
                repository: 'petclinic', version: '6.2.8'
            }
        }
        //stage 6: To host the war in the tomcat
        stage('tomcat') {
            steps {
                sh 'scp target/petclinic.war mailto:ubuntu@172.31.25.155:/tmp'
                sh 'ssh mailto:ubuntu@172.31.25.155 sudo mv /tmp/petclinic.war  /var/lib/tomcat10/webapps/'
                sh 'ssh mailto:ubuntu@172.31.25.155 sudo chown tomcat:tomcat /var/lib/tomcat10/webapps/petclinic.war'
                sh 'ssh mailto:ubuntu@172.31.25.155 sudo systemctl restart tomcat10'
            }
        }
    }
}
